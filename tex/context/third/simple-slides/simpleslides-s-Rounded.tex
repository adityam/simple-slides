%D We define our colorscheme:

\definecolor [simpleslides:backgroundcolor]  [s=.95]
\definecolor [simpleslides:contrastcolor]    [r=.58,g=.81,b=.58]
\definecolor [simpleslides:textcolor]        [r=.09,g=.2,b=.41]
\definecolor [simpleslides:variantcolor]     [r=.04,g=.4,b=.4]
\definecolor [simpleslides:itemize:color]    [simpleslides:textcolor]

%D Textcolor is blue:

\setupcolors[textcolor=simpleslides:textcolor]

%D We let Metapost calculate the background:

\startuseMPgraphic{simpleslides:MP:horizontal}
StartPage ;
fill Page withcolor \MPcolor{simpleslides:backgroundcolor} ;
numeric a ; a = 2cm ;
numeric b ; b = 0.7cm ;
numeric c ; c = 6cm ;
numeric d ; d = .7cm ;
pair zc[] ;
zc1 = llcorner Page shifted (a,0) ;
zc2 = ulcorner Page shifted (a,-a-b) ;
zc3 = ulcorner Page shifted (a+b/4,-a-b/4) ;
zc4 = ulcorner Page shifted (a+b,-a) ;
zc5 = urcorner Page shifted (0,-a) ;
zc6 = ulcorner Page shifted (c,0) ;
zc7 = ulcorner Page shifted (c,-a) ;
path pa[] ;
pa[1] = llcorner Page -- zc1 -- zc2 .. zc3 .. zc4 -- zc7 -- zc6 -- ulcorner Page -- cycle ;
fill pa[1] withcolor \MPcolor{simpleslides:contrastcolor} ;
draw \sometxt{\framed[frame=off,width=2cm,height=2cm]{\color[simpleslides:backgroundcolor]{\pagenumber}}} ;
StopPage ;
\stopuseMPgraphic

\startuniqueMPgraphic{simpleslides:MP:ornament}
StartPage ;
numeric a ; a = 2cm ;
numeric b ; b = 0.7cm ;
numeric c ; c = 6cm ;
numeric d ; d = .7cm ;
pair zc[] ;
zc8 = ulcorner Page shifted (a/2,-2.2*a) ;
zc9 = zc8 shifted (0,-d) ;
zc10 = urcorner Page shifted (-a,-2.2*a-d) ;
zc11 = zc10 shifted (0,d) ;
zc12 = zc8 shifted (-d/2,-d/2) ;
path pa[] ;
pa[3] = zc8 .. zc12 .. zc9 -- zc10 -- zc11 -- cycle ;
fill pa[3] withcolor \MPcolor{simpleslides:textcolor} ;
StopPage ;
\stopuniqueMPgraphic

\startuniqueMPgraphic{simpleslides:MP:title}
StartPage ;
numeric a ; a = 4cm ;
numeric b ; b = 3cm ;
numeric c ; c = 8cm ;
numeric d ; d = .7cm ;
path pa[] ;
pair zc [] ;
fill Page withcolor \MPcolor{simpleslides:backgroundcolor} ;
pa[1] = ulcorner Page -- ulcorner Page shifted (PaperWidth/2,0) -- llcorner Page shifted (PaperWidth/2,0) -- llcorner Page -- cycle ;
fill pa[1] withcolor \MPcolor{simpleslides:contrastcolor} ;
zc1 = ulcorner Page shifted (PaperWidth/2,-b) ;
zc2 = zc1 shifted (-c,0) ;
zc3 = zc2 shifted (0,-a) ;
zc4 = zc3 shifted (c,0) ;
zc5 = zc2 shifted (-1.5cm,-a/2) ;
pa[2] = zc1 -- zc2 .. zc5 .. zc3 -- zc4 -- cycle ;
fill pa[2] withcolor \MPcolor{simpleslides:backgroundcolor} ;
zc6 = llcorner Page shifted (PaperWidth/2,0) ;
zc7 = 1/2[zc6,zc4] ;
zc8 = zc7 shifted (-.75*b,d/2) ;
zc9 = zc8 shifted (0,-d) ;
zc10 = zc9 shifted (1.3*c,0) ;
zc11 = zc10 shifted (0,d) ;
zc12 = zc10 shifted (d/2,d/2) ;
pa[3] = zc8 -- zc9 -- zc10 .. zc12 .. zc11 -- cycle ;
fill pa[3] withcolor \MPcolor{simpleslides:textcolor} ;
StopPage ;
\stopuniqueMPgraphic
